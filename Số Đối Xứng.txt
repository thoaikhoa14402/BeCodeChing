#include<iostream>

using namespace std;

bool isPalindrome(int n) {
	int num = 0, temp = n;
	while (n != 0) {
		num = num * 10 + (n % 10);
		n /= 10;
	}
	if (num == temp) return true;
	return false;
}
int main() {
	int n;
	cout << "Enter n: ";
	cin >> n;
	if (isPalindrome(n)) cout << n << " la so doi xung";
	else cout << n << " khong phai so doi xung";
	return 0;
}
